error to seperate file:
 
  <property name="FILE_PATTERN" value="%d{dd-MM-yyyy HH:mm:ss.SSS} %-5level [%thread] %logger{36} - %msg%n"/>
  <property name="CONSOLE_PATTERN" value="%green(%d{dd-MM-yyyy HH:mm:ss.SSS}) %highlight(%-5level) %magenta([%thread]) %cyan(%logger{36}) - %msg%n"/>
  <property name="USER_LOG_HOME" value="logs" />

  <appender name="FILE_APPENDER_INFO" class="ch.qos.logback.core.rolling.RollingFileAppender">
    <file>${USER_LOG_HOME}/app.log</file>
    <rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
      <fileNamePattern>${USER_LOG_HOME}/app-%d{yyyyMMdd}.log</fileNamePattern>
      <maxHistory>30</maxHistory>	 <!-- keep 30 days' worth of history -->
    </rollingPolicy>
    <encoder>
      <pattern>${FILE_PATTERN}</pattern>
    </encoder>
  </appender>

  <appender name="FILE_APPENDER_ERROR" class="ch.qos.logback.core.rolling.RollingFileAppender">
    <file>${USER_LOG_HOME}/app-error.log</file>
    <rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
      <fileNamePattern>${USER_LOG_HOME}/app-error-%d{yyyyMMdd}.log</fileNamePattern>
      <maxHistory>30</maxHistory>
    </rollingPolicy>
    <encoder>
      <pattern>${FILE_PATTERN}</pattern>
    </encoder>
    <filter class="ch.qos.logback.classic.filter.ThresholdFilter">
        <level>ERROR</level>  <!-- Only write error log to app-error.log file -->
    </filter>
  </appender>

  <appender name="FILE_APPENDER_QUERY" class="ch.qos.logback.core.rolling.RollingFileAppender">
    <file>${USER_LOG_HOME}/app-query.log</file>
    <rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
      <fileNamePattern>${USER_LOG_HOME}/app-query-%d{yyyyMMdd}.log</fileNamePattern>
      <maxHistory>30</maxHistory>
    </rollingPolicy>
    <encoder>
      <pattern>${FILE_PATTERN}</pattern>
    </encoder>
  </appender>
  
  <!-- show hibernate sql -->
  <logger name="org.hibernate.SQL" level="DEBUG" additivity="false">
    <appender-ref ref="STDOUT"/>
    <appender-ref ref="FILE_APPENDER_QUERY"/>
  </logger>

  <!-- Show hibernate query params -->
  <logger name="org.hibernate.type.descriptor.sql.BasicExtractor" level="TRACE" additivity="false">
    <appender-ref ref="FILE_APPENDER_QUERY"/>
  </logger>

  <root level="INFO">
    <appender-ref ref="STDOUT"/>
    <appender-ref ref="FILE_APPENDER_INFO"/>
    <appender-ref ref="FILE_APPENDER_ERROR"/>
  </root>
